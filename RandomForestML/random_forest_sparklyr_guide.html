Class imbalance is common in churn — typically 15–25% churn rate. Spark RF offers several native strategies plus manual approaches.
Strategy 1 · class weights (native, recommended first)
# Add a weight column: minority class gets higher weight
n_total <- sdf_nrow(train)
n_churn <- train %>% filter(label == 1) %>% sdf_nrow()
n_active <- n_total - n_churn

# Inverse-frequency weights
w_churn  <- n_total / (2 * n_churn)
w_active <- n_total / (2 * n_active)

train_weighted <- train %>%
  mutate(
    class_weight = ifelse(label == 1, w_churn, w_active)
  )

# Pass weight column to the model
rf_model_w <- ml_random_forest_classifier(
  x              = train_weighted,
  formula        = rf_formula,
  num_trees      = 200,
  max_depth      = 8,
  weight_col     = "class_weight",   # ← key argument
  seed           = 42
)
Strategy 2 · oversampling minority in Spark
# Separate classes
majority <- train %>% filter(label == 0)
minority  <- train %>% filter(label == 1)

# Oversample minority to match majority size
ratio <- ceiling(sdf_nrow(majority) / sdf_nrow(minority))
minority_over <- sdf_sample(
  minority,
  fraction = ratio,
  replacement = TRUE,
  seed = 42
)

train_balanced <- sdf_bind_rows(majority, minority_over)

# Shuffle to avoid order bias
train_balanced <- train_balanced %>%
  sdf_sample(fraction = 1.0, replacement = FALSE, seed = 99)
Strategy 3 · undersampling majority
# Downsample majority to minority size
target_n    <- sdf_nrow(minority)
under_frac  <- target_n / sdf_nrow(majority)

majority_under <- sdf_sample(
  majority,
  fraction    = under_frac,
  replacement = FALSE,
  seed        = 42
)

train_under <- sdf_bind_rows(majority_under, minority)
Strategy 4 · threshold tuning (post-model)
# Default threshold = 0.5 — move it to favour recall for Churn
preds_raw <- ml_predict(rf_model, test)

preds_threshold <- preds_raw %>%
  mutate(
    prob_churn     = vector_to_array(probability)[2],
    pred_adjusted  = ifelse(prob_churn >= 0.35, 1, 0)
    # Lower threshold → catches more churners (higher recall)
    # Raise threshold → fewer false alarms (higher precision)
  )
When to use what
class_weightMild imbalance (<10:1)
oversampleSmall minority class
undersampleHuge majority, data rich
thresholdPost-hoc precision/recall trade
Metrics to prefer
AUC-ROCAlways useful
AUC-PRImbalanced datasets
F1-ScoreBalance P & R
AccuracyMisleading — avoid
⚠️ Spark's Random Forest does not expose L1/L2 penalty terms (those are for linear models). Regularisation for RF is done via structural constraints and subsampling.
Structural regularisation parameters
# All regularisation knobs in one model call
rf_reg <- ml_random_forest_classifier(
  x = train,
  formula = rf_formula,

  # ── Tree complexity (depth controls over-fitting) ──────
  num_trees           = 300,   # more trees ≠ overfit (just slower)
  max_depth           = 6,    # key lever: deeper = more overfit
  max_bins            = 32,   # binning resolution for cont. features

  # ── Minimum samples in a node (prevents tiny leaves) ──
  min_instances_per_node = 10,  # ↑ = more regularised
  min_info_gain          = 0.01, # split only if gain > this

  # ── Subsampling (bagging + feature randomness) ────────
  subsampling_rate         = 0.7,   # row sampling per tree
  feature_subset_strategy  = "sqrt", # col sampling: "sqrt","log2","auto"

  seed = 42
)
Parameter effects cheatsheet
max_depth ↓Less overfit, simpler trees
min_instances_per_node ↑Prunes small splits
min_info_gain ↑Only splits that matter
subsampling_rate ↓More diverse trees, less variance
feature_subset_strategy"sqrt" for classification
num_trees ↑Stabilises predictions (no overfit risk)
Pre-processing regularisation (via recipes in R)
# Collect sample to R, fit recipe, push features back
train_r <- collect(sample_n(train, 50000))

rec <- recipe(label ~ ., data = train_r) %>%
  step_nzv(all_predictors()) %>%        # remove near-zero variance
  step_corr(all_numeric_predictors(),   # remove highly correlated
            threshold = 0.90) %>%
  step_impute_median(all_numeric_predictors()) %>%
  step_dummy(all_nominal_predictors())

rec_trained <- prep(rec, training = train_r)

Grid search with cross-validation
# ── 1. Define the pipeline ───────────────────────────
pipeline <- ml_pipeline(sc) %>%
  ft_r_formula(rf_formula) %>%
  ml_random_forest_classifier(
    label_col   = "label",
    features_col = "features"
  )

# ── 2. Hyperparameter grid ───────────────────────────
param_grid <- list(
  random_forest_classifier = list(
    num_trees  = c(100L, 200L, 300L),
    max_depth  = c(4L, 6L, 8L),
    min_instances_per_node = c(5L, 10L),
    subsampling_rate = c(0.7, 0.9)
  )
)

# ── 3. Evaluator ─────────────────────────────────────
evaluator <- ml_binary_classification_evaluator(
  sc,
  label_col    = "label",
  metric_name  = "areaUnderROC"
)

# ── 4. Cross-validation (3-fold) ─────────────────────
cv <- ml_cross_validator(
  sc,
  estimator           = pipeline,
  estimator_param_maps = param_grid,
  evaluator           = evaluator,
  num_folds           = 3,
  seed                = 42
)

cv_model <- ml_fit(cv, train)

Score new dataset & join back
# ── 1. Load new data ─────────────────────────────────
new_data <- spark_read_csv(
  sc,
  name   = "new_customers",
  path   = "/data/new_customers.csv",
  header = TRUE
)

# Apply same feature engineering as training
new_data <- new_data %>%
  mutate(
    tenure_sq    = tenure * tenure,
    avg_charges  = TotalCharges / pmax(tenure, 1)
  )

# ── 2. Score ─────────────────────────────────────────
scored <- ml_predict(rf_loaded, new_data)

# ── 3. Extract probability & binary prediction ───────
scored_clean <- scored %>%
  mutate(
    prob_churn  = vector_to_array(probability)[2],
    # Binary label: "Churn" or "Active"
    Prediction  = ifelse(prediction == 1, "Churn", "Active")
  )

# ── 4. Join back to original dataset ─────────────────
# Keep original columns + add Prediction column
final_df <- new_data %>%
  left_join(
    scored_clean %>%
      select(CustomerID, prob_churn, Prediction),
    by = "CustomerID"
  )

# Peek at result
final_df %>%
  select(CustomerID, tenure, MonthlyCharges,
         prob_churn, Prediction) %>%
  head(10) %>%
  collect()
